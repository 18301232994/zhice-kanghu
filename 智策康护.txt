        const evaluation = (raw.match(/评估建议([\s\S]*?)$/i) || [])[1] || '';
        renderPlan({ goals: goals.trim(), activities: activities.trim(), communication: communication.trim(), evaluation: evaluation.trim() });
      } catch (e) {
        console.warn('使用本地预设方案', e);
        const plan = PRESET_PLANS[profession] || PRESET_PLANS.generic;
        renderPlan(plan);
      } finally {
        hideLoading();
      }
    }

    btn.addEventListener('click', handleGenerate);
  })();
